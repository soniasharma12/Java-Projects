# Create the 'get_binary_function()'.
def get_binary_digits(dividend):
  binary_digits=[]
  while dividend != 0:
    quot = dividend // 2
    remainder = dividend % 2
    binary_digits.append(remainder)
    dividend = quot
  binary_digits.reverse()
  return binary_digits
